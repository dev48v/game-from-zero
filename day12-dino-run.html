    A canvas and the world
    
        
    One canvas, a ground line, and the variables every frame will touch: the dino, the list of cacti, the score, and a speed that grows.
    
        
    
          
    <canvas id="c" width="600" height="200"></canvas>
<script>
const c = document.getElementById("c"), ctx = c.getContext("2d");
const GROUND = 170;          // y of the ground line
let dino, cacti, score, speed, over, raf;
</script>
    
          
        
    
      
    2. 

      
  2. 
        
    2
    Reset puts everyone home
    
        
    The dino is a box sitting on the ground with no vertical speed. Cacti start empty, score is zero, speed is gentle. We call reset both at startup and on restart.
    
        
    
          
    function reset(){
  dino  = { x:40, y:GROUND-44, w:24, h:44, vy:0 };
  cacti = []; score = 0; speed = 4; over = false;
  spawn();                       // queue the first cactus
  cancelAnimationFrame(raf); loop();
}
    
          
        
    
      
    3. 

      
  3. 
        
    3
    Gravity and the jump
    
        
    A jump just sets an upward velocity. Every frame gravity pulls that velocity down and adds it to the dino's y. When the dino lands on the ground we clamp it and stop. Only allow a jump while grounded so you can't fly.
    
        
    
          
    function jump(){
  if (dino.y >= GROUND - dino.h) dino.vy = -11;   // grounded only
}
function physics(){
  dino.vy += 0.6;                 // gravity
  dino.y  += dino.vy;
  if (dino.y > GROUND - dino.h){ dino.y = GROUND - dino.h; dino.vy = 0; }
}
    
          
        
    
      
    4. 

      
  4. 
        
    4
    Spawn and scroll the cacti
    
        
    A cactus is a box that starts off the right edge. Each frame every cactus slides left by speed. When one passes a gap, queue the next at a random distance. Drop cacti that have left the screen.
    
        
    
          
    function spawn(){
  const h = 28 + Math.random()*22;
  cacti.push({ x:600 + Math.random()*120, y:GROUND-h, w:16, h });
}
function move(){
  cacti.forEach(o => o.x -= speed);
  cacti = cacti.filter(o => o.x + o.w > 0);    // off-screen → gone
  const last = cacti[cacti.length-1];
  if (!last || last.x < 600 - (160 + Math.random()*160)) spawn();
}
    
          
        
    
      
    5. 

      
  5. 
        
    5
    Collision = two boxes overlapping
    
        
    Axis-Aligned Bounding Box: two rectangles touch only if they overlap on both x and y at once. If the dino's box overlaps any cactus box, the run is over.
    
        
    
          
    function hits(a, b){
  return a.x < b.x + b.w && a.x + a.w > b.x &&
         a.y < b.y + b.h && a.y + a.h > b.y;
}
// in the loop:
if (cacti.some(o => hits(dino, o))) over = true;
    
          
        
    
      
    6. 

      
  6. 
        
    6
    Score, speed-up, and the loop
    
        
    Score climbs every frame with distance; nudge the speed up slowly so it never stops getting harder. The loop ties it together: physics, move, collide, draw, repeat — and bail out when over.
    
        
    
          
    function loop(){
  if (over){ draw(); return; }      // freeze on game over
  physics(); move();
  if (cacti.some(o => hits(dino, o))) { over = true; }
  score += 1; speed += 0.002;       // distance + difficulty
  draw();
  raf = requestAnimationFrame(loop);
}
    
          
        
    
      
    7. 

      
  7. 
        
    7
    Draw the scene
    
        
    Clear, draw the ground line, draw the dino, draw each cactus, and print the score. Canvas is immediate-mode: nothing persists, so you repaint the whole world every frame.
    
        
    
          
    function draw(){
  ctx.fillStyle = "#020617"; ctx.fillRect(0,0,600,200);
  ctx.strokeStyle = "#334155"; ctx.beginPath();
  ctx.moveTo(0,GROUND+1); ctx.lineTo(600,GROUND+1); ctx.stroke();
  ctx.fillStyle = "#10b981";
  ctx.fillRect(dino.x, dino.y, dino.w, dino.h);          // dino
  ctx.fillStyle = "#22c55e";
  cacti.forEach(o => ctx.fillRect(o.x, o.y, o.w, o.h));  // cacti
  ctx.fillStyle = "#e2e8f0"; ctx.font = "14px monospace";
  ctx.fillText("Score " + score, 12, 22);
}

    Wire input and start
    
        
    Space and ↑ jump (and restart when over); a tap does the same. Then call reset() once and the runner is live. Double-click the file — no server needed.
    
        
    
          
    addEventListener("keydown", e => {
  if (e.code === "Space" || e.key === "ArrowUp"){
    e.preventDefault();
    over ? reset() : jump();
  }
});
c.addEventListener("pointerdown", () => over ? reset() : jump());
reset();
