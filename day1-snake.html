🔨 Build it yourself

    
Estimated time: 20 minutes. No npm, no build. One HTML file.

    
    
      
  1. 
        
    1
    Make an empty file
    

    mkdir my-snake
cd my-snake
touch index.html
    
        
    Windows: right-click → New → Text File → rename to index.html
    
      
    2. 
      
  2. 
        
    2
    Paste this in index.html
    
        
    Entire game in one file. ~80 lines of JS.
    
        
    

    <!DOCTYPE html>
<html><head><title>Snake</title></head>
<body style="background:#0f172a;color:#e2e8f0;text-align:center;font-family:system-ui;">
  <h1>🐍 Snake</h1>
  <div>Score: <span id="score">0</span></div>
  <canvas id="c" width="400" height="400" style="background:#020617;border:2px solid #334155;"></canvas>
  <p>Arrow keys to move · Space to restart</p>
<script>
const canvas = document.getElementById("c");
const ctx = canvas.getContext("2d");
const scoreEl = document.getElementById("score");
const GRID = 20, SIZE = canvas.width / GRID;
let snake, dir, nextDir, food, score, dead;

function reset() {
  snake = [{x:10,y:10},{x:9,y:10},{x:8,y:10}];
  dir = {x:1,y:0}; nextDir = dir;
  food = randCell(); score = 0; dead = false;
  scoreEl.textContent = score;
}
function randCell() {
  while (true) {
    const c = { x: Math.floor(Math.random()*GRID), y: Math.floor(Math.random()*GRID) };
    if (!snake.some(s => s.x===c.x && s.y===c.y)) return c;
  }
}
function step() {
  if (dead) return;
  dir = nextDir;
  const head = { x: snake[0].x + dir.x, y: snake[0].y + dir.y };
  if (head.x<0 || head.y<0 || head.x>=GRID || head.y>=GRID) return die();
  if (snake.some(s => s.x===head.x && s.y===head.y)) return die();
  snake.unshift(head);
  if (head.x===food.x && head.y===food.y) { score++; scoreEl.textContent = score; food = randCell(); }
  else snake.pop();
}
function die() {
  dead = true;
  ctx.fillStyle = "rgba(220,38,38,.4)";
  ctx.fillRect(0,0,canvas.width,canvas.height);
  ctx.fillStyle = "#fff"; ctx.font = "bold 32px system-ui"; ctx.textAlign = "center";
  ctx.fillText("Game Over", canvas.width/2, canvas.height/2);
}
function draw() {
  if (dead) return;
  ctx.fillStyle = "#020617"; ctx.fillRect(0,0,canvas.width,canvas.height);
  ctx.fillStyle = "#ef4444"; ctx.fillRect(food.x*SIZE+2, food.y*SIZE+2, SIZE-4, SIZE-4);
  snake.forEach((s,i)=>{ ctx.fillStyle = i===0 ? "#22c55e" : "#16a34a"; ctx.fillRect(s.x*SIZE+1, s.y*SIZE+1, SIZE-2, SIZE-2); });
}
addEventListener("keydown", e => {
  const k = e.key;
  if (k === " ") { reset(); return; }
  if (k==="ArrowUp"    && dir.y!== 1) nextDir = {x:0,y:-1};
  if (k==="ArrowDown"  && dir.y!==-1) nextDir = {x:0,y: 1};
  if (k==="ArrowLeft"  && dir.x!== 1) nextDir = {x:-1,y:0};
  if (k==="ArrowRight" && dir.x!==-1) nextDir = {x: 1,y:0};
});
reset();
setInterval(()=>{ step(); draw(); }, 100);
</script></body></html>
