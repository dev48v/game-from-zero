    Slide + merge ONE row (the whole game)
    

    function slide(row) {
  let r = row.filter(v => v);          // drop zeros
  for (let i = 0; i < r.length - 1; i++)
    if (r[i] === r[i+1]) { r[i] *= 2; r[i+1] = 0; } // merge once
  r = r.filter(v => v);                // drop the merged-away zeros
  while (r.length < 4) r.push(0);      // pad back to 4
  return r;
}
