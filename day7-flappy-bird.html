    // Spawn a new pipe pair every PIPE_INTERVAL frames
if (frame % PIPE_INTERVAL === 0) {
  const gapY = 80 + Math.random() * (H - 200);
  pipes.push({ x: W, gapY });
}
// Each frame, slide all pipes left
pipes.forEach(p => p.x -= SPEED);

    AABB collision detection
    // Axis-Aligned Bounding Box: do two rectangles overlap?
function hits(p) {
  const bx = bird.x - bird.r, by = bird.y - bird.r;
  const bw = bird.r * 2,      bh = bird.r * 2;
  // top pipe rect
  if (bx < p.x+PIPE_W && bx+bw > p.x &&
      by < p.gapY - GAP/2) return true;
  // bottom pipe rect
  if (bx < p.x+PIPE_W && bx+bw > p.x &&
      by+bh > p.gapY + GAP/2) return true;
  return false;
}

  5. 5
    Score + game loop with requestAnimationFrame
    // Score increments when bird clears a pipe's x position
pipes.forEach(p => {
  if (!p.scored && p.x + PIPE_W < bird.x) {
    score++; p.scored = true;
  }
});

// Game loop — browser calls this ~60 times per second
function loop() {
  update();
  draw();
  if (state !== "dead") requestAnimationFrame(loop);
}
